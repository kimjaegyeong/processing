#### 1. 화면에 색이 다른 원 3개를 만들어 빨간색 원은 좌->우, 초록색 원은 위->아래, 파란색 원은 아래->위로 움직이도록 하기.
 
```
void setup(){
 
  size(1000,1000); // 실행창의 크기를 지정한다.

}

int i;
int j;
int k=1000; // i,j,k 는 정수형 변수로 선언하고, 이것은 후에 원의 이동에 관여한다. k는 1000의 값을 넣어 초기화한다.
void draw(){
  background(0); //배경색을 지정한다.
  fill(255,0,0); // 원의 색을 빨간색으로 지정한다.
  ellipse(i,160,300,300); // x좌표를 i로 두고, y좌표를 160, 원의 크기는 가로 세로 300씩 지정한다.
  i=i+3; // i는 루프를 돌 때마다 3씩 증가한다.
  if(i>width+160){i=0;} // 만약 i가 실행창의 너비크기+160의 값보다 커지면 i는 0으로 초기화한다.
  fill(0,255,0); // 원의 색을 초록색으로 지정한다.
  ellipse(160,j,300,300); // x좌표를 160, y좌표를 j, 원의 크기는 가로 세로 300씩 지정한다.
  j=j+3; // j 역시 루프를 돌 때마다 3씩 증가한다.
  if(j>height+160){ j=0;} // j가 실행창의 높이크기+160의 값보다 커지면 j는 0으로 초기화한다.
  
fill(0,0,255); // 원의 색을 파란색으로 지정한다.
  ellipse(width-160,k,300,300); // x좌표는 실행창의 너비-160, y 좌표는 k, 원의 크기는 가로 세로 300으로 지정한다.
  k=k-3; // 루프를 돌 때마다 k의 값은 3씩 작아진다.
  if(k<-160){ k=1000;} // k의 값이 -160보다 작아지면 k는 1000으로 초기화한다.
}
```

#### 2.문자 출력하기, 키보드 사용해서 색상 바꾸기
```
PFont f; // PFont형식의 f를 선언한다.
void setup(){
size(800,300);
f = createFont("굴림",50); // f는 굴림체의 크기 50을 갖는 문자열로 초기화된다.
textFont(f); // 만들어질 문자열은 f의 설정을 따른다. 
}
int i, dir=1; // 좌우 움직임의 증가를 맡을 i와, 우->좌로 움직일 때 필요한 dir을 정수형 변수로 선언한다. dir은 1의 값을 넣어 초기화한다.
void draw(){
int R=0; 
int G=0;
int B=0; // R,G,B는 색을 지정할 때 사용할 정수형 변수이다. 0의 값을 넣어 초기화한다.
background(0,230,180); //배경색을 지정한다.
fill(0); //문자열의 색을 지정한다. 
if(keyPressed){ //만약 키가 눌렸을 때,
  if(key=='r') R=255; //키보드의 r이 눌렸다면 R의 값이 255이고
  if(key=='b') B=255; //키보드의 b가 눌렸다면 B의 값이 255이고
  if(key=='g') G=255; //키보드의 g가 눌렸다면 B의 값이 255이다. 
fill(R,G,B);} // 문자열의 색을 R,G,B에 따라 지정한다. 

text("컴퓨터공학과 화이팅~",i,150); // 문자열의 내용을 지정하고, x좌표는 i, y좌표는 150으로 지정한다.
 if(i>width) dir=-1; // 만약 i의 값이 실행창 너비 크기보다 크다면 dir의 값은 -1이고,

 if(i<0) dir=1; // i가 0보다 작아진다면 dir의 값은 1이된다.

 i = i+dir; // i는 i+dir의 값을 저장한다. 
}
```

#### 3. Transparency (원본 그림 위에 투명도가 적용 된 사진 입히기)
```
PImage img;
float offset =0; 
float easing = 0.05; //마우스를 움직일 때 투명화 된 사진이 따라오는 속도이다.

void setup() {
  size(720 , 960);
  img = loadImage("cherryblossom.png");  //이미지를 로드하여 img에 저장한다 
}

void draw() { 
  image(img, 0, 0); // 이미지의 기본 좌표를 설정하여 준다.  
  float dx = (mouseX-img.width/2)- offset ; //투명화된 사진이 마우스 좌표를 따라가되, 원래사진의 반 이상 이탈하지 않도록 한다.
 offset += dx * easing; // offset을 dx*esasing 만큼 증가시켜  투명화 된 사진이 마우스의 좌표에 따라 움직인다.
  tint(255, 80); //  투명도를 지정한다.
  image(img, offset, 0); // 투명화 된 사진의 x, y좌표를 설정하여 준다.
}
```

#### 4. Interactivity ( 마우스의 좌표에 따라 사각형이 생성되고, 사라진다)
 ```
void setup() {
  size(100, 100);
  noStroke(); //테두리가 없도록 한다. 
  fill(0); //검정색을 입힌다. 
}


void draw() {
  background(204);
  if (mouseX < 33) { // 마우스의 X좌표가 33보다 적을 때 
    rect(0, 0, 33, 100); // 0,0을 기준으로 가로 33, 세로 100인 사각형을 만든다.
  } else if (mouseX < 66) { //마우스의 x좌표가 66보다 적을 때
    rect(33, 0, 33, 100);  // 33, 0 기준으로 가로 33, 세로 100인 사각형을 만든다.
  } else {
    rect(66, 0, 33, 100); // 위의 경우에 모두 해당이 안 될 시, 66,0 기준으로 가로 33, 세로 100인 사각형을 만든다.
  }
}
```

#### 5. Create Graphics( 본래 화면과, 그래픽스 화면 안에서의 동작을 지정한다. ) 
```
PGraphics pg; 

void setup() {
  size(640, 360);
  pg = createGraphics(400, 200); // 가로로 400, 세로로 200 크기인 그래픽스를 생성한다.  
}



void draw() {
  fill(0,12); // 0은 불투명도, 12는 흰색과 검은색 사이의 색을 숫자로 표현한 것이다.
  rect(0, 0, width, height); //배경이 되는 사각형이다. 
  fill(255); // 원의 색을 지정한다.
  noStroke(); // 원의 테두리를 삭제한다. 
  ellipse(mouseX, mouseY, 60, 60); //원을 생성하며, 원의 좌표는 마우스의 x,y좌표를 따르게한다, 원의 크기는 60,60 이다.
  
  pg.beginDraw(); // 그래픽스를 사용하기 위해 선언한다.
  pg.background(51); // 그래픽스의 배경 색을 지정한다. 
  pg.noFill(); //원의 내부를 투명하게 만든다. 
  pg.stroke(255); // 원의 테두리 색을 지정한다.
  pg.ellipse(mouseX-120, mouseY-60, 60, 60); // 원의 좌표와, 크기를 설정한다.
  pg.endDraw(); // 그래픽스 사용을 끝마쳤음을 알리기 위해 선언한다. 
  
  
  image(pg, 120, 60); // x좌표 120, y좌표 60에 pg을 삽입한다. 
}
```
